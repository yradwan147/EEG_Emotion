% Inline comment:  \kilich{my note here}
%   Multi-paragraph safe (uses \color switch + \bfseries instead of
%   \textcolor + \textbf, both of which forbid \par in arguments).
\long\newcommand{\kilich}[1]{%
  \ifshowcomments%
    \begingroup
      \color{kilichcolor}\bfseries
      [K: #1]%
    \endgroup
  \fi%
}

% Tracked-change replacement:  \kilichchg{old text}{new text}
%   Note: \sout cannot span paragraphs, so OLD text must be inline only.
%   For multi-paragraph replacements, use \kilich on the new text and
%   delete (or comment out) the old text directly.
\newcommand{\kilichchg}[2]{%
  \ifshowcomments%
    {\color{red}\sout{#1}}\,{\color{kilichcolor}\bfseries #2}%
  \else%
    #2%
  \fi%
}

% Margin-note version:  \kilichmn{my margin comment}
\long\newcommand{\kilichmn}[1]{%
  \ifshowcomments%
    \marginpar{%
      \raggedright\footnotesize
      \begingroup\color{kilichcolor}\bfseries[K]\,#1\endgroup%
    }%
  \fi%
}
